#install.packages("gbm")
library(gbm)
?gbm
gbm1 <- gbm(released ~., data = Arrests, distribution = "bernoulli",
            n.trees = 500, interaction.depth = 5, n.minobsinnode = 1,
            bag.fraction = 0.5, train.fraction = .75, n.cores = 8)
best.iter <- gbm.perf(gbm1, method = "OOB")
